# useActivateOnEnter Custom Hook

## Imports

```tsx
import * as React from 'react'
import cx from 'classnames'
import { LabelTitle } from '@src/components/form-components/utils/utility-components/label/index.tsx'
import { HelpText } from @src/components/form-components/utils/utility-components/help/index.tsx'
import { useActivateOnEnter } from '@src/components/form-components/utils/utility-components/hooks/useActivateOnEnter/index.tsx'
```

## Props

```tsx
type TInputProps = React.InputHTMLAttributes<HTMLInputElement> & {
  className?: String
  label?: String
  help?: String
  labelClassName?: String
  inputClassName?: String
  helpClassName?: String
  activateEnterPress?: Boolean
  clearOnEnterPress?: Boolean
  onEnterPress?: () => void
  onChange?: (event: React.ChangeEvent<HTMLInputElement>) => void
  value?: String
}
```

## InputTest Component

```tsx
// eslint-disable-next-line react/display-name
export const InputTest = React.forwardRef<HTMLInputElement, TInputProps>(
  (
    {
      label,
      help,
      className,
      labelClassName,
      inputClassName,
      helpClassName,
      activateEnterPress,
      clearOnEnterPress,
      onEnterPress,
      onChange,
      value: initialValue,
      ...rest
      // eslint-disable-next-line no-unused-vars, @typescript-eslint/no-unused-vars
    },
    ref
  ) => {
    const { value, setValue, inputRef, handleKeyDown } = useActivateOnEnter({
      activateEnterPress,
      clearOnEnterPress,
      onEnterPress,
      onChange,
      value: initialValue,
    })

    return (
      <div className={cx('input-container', className)}>
        <LabelTitle label={label} className={cx('label-title', labelClassName)} />
        <input
          ref={inputRef}
          className={cx('input-container__text', inputClassName)}
          value={value}
          onChange={e => setValue(e.target.value)}
          onKeyDown={handleKeyDown}
          {...rest}
        />
        <HelpText className={cx('help-text', helpClassName)} help={help} />
      </div>
    )
  }
)
```

## Usage of **_clearOnEnterPress_**

It will clear the input field on `Enter Press`

`<InputTest label='just a title' help='some helper text' clearOnEnterPress />`

## Usage of **_activateEnterPress_**

It will submit the input field on `Enter Press`

`<InputTest label='just a title' help='some helper text' activateEnterPress />`

## Usage of **_onEnterPress_**

It will submit the input field on `Enter Press`

`<InputTest label='just a title' help='some helper text' activateEnterPress onEnterPress=(submit())/>`
